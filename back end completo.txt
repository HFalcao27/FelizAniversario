1️⃣ BACKEND – CRIAR A API REST
📁 Controller – AniversarioController.java

java-

@RestController
@RequestMapping("/api/aniversarios")
@CrossOrigin(origins = "*") // libera o front
public class AniversarioController {

    private final AniversarioRepository repository;

    public AniversarioController(AniversarioRepository repository) {
        this.repository = repository;
    }

    // 🔹 LISTAR
    @GetMapping
    public List<Aniversario> listar() {
        return repository.findAll();
    }

    // 🔹 SALVAR
    @PostMapping
    public Aniversario salvar(@RequestBody Aniversario aniversario) {
        return repository.save(aniversario);
    }

    // 🔹 EDITAR
    @PutMapping("/{id}")
    public Aniversario editar(
            @PathVariable Long id,
            @RequestBody Aniversario dados) {

        Aniversario a = repository.findById(id)
                .orElseThrow();

        a.setNome(dados.getNome());
        a.setTelefone(dados.getTelefone());
        a.setMensagem(dados.getMensagem());
        a.setDataAniversario(dados.getDataAniversario());
        a.setHoraEnvio(dados.getHoraEnvio());

        return repository.save(a);
    }

    // 🔹 REMOVER
    @DeleteMapping("/{id}")
    public void remover(@PathVariable Long id) {
        repository.deleteById(id);
    }
}

function listarAniversarios() {
    fetch(API_URL)
        .then(res => res.json())
        .then(aniversarios => {
            lista.innerHTML = "";

            aniversarios.forEach(item => {
                const li = document.createElement("li");

                li.innerHTML = `
                    <span>
                        <strong>${item.nome}</strong><br>
                        🎂 ${formatarData(item.dataAniversario)}<br>
                        📞 ${item.telefone}<br>
                        💬 ${item.mensagem || ""}
                    </span>
                    <div>
                        <button onclick="enviarWhatsApp('${item.nome}', '${item.telefone}', '${item.mensagem || ""}')">📲</button>
                        <button onclick="editar(${item.id})">✏️</button>
                        <button onclick="remover(${item.id})">🗑️</button>
                    </div>
                `;

                lista.appendChild(li);
            });
        });
}

5️⃣ FRONT-END – REMOVER (DELETE)

function remover(id) {
    if (!confirm("Deseja remover este aniversário?")) return;

    fetch(`${API_URL}/${id}`, {
        method: "DELETE"
    })
    .then(() => listarAniversarios());
}

function marcarAniversarios() {
    fetch(API_URL)
        .then(res => res.json())
        .then(eventos => {
            eventos.forEach(evento => {
                const data = new Date(evento.dataAniversario);
                if (
                    data.getMonth() === mesAtual &&
                    data.getFullYear() === anoAtual
                ) {
                    document.querySelectorAll(".days div")
                        .forEach(dia => {
                            if (Number(dia.textContent) === data.getDate()) {
                                dia.style.backgroundColor = "#ffcc00";
                            }
                        });
                }
            });
        });
}
